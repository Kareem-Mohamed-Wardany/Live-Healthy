@startuml
actor       Administrator       as User
boundary    AdministratorPage  as PatientPage
control     SystemHandler     as SystemHandler
entity      Administrator as Administrator



User -> PatientPage : Verify Doctor
PatientPage -> SystemHandler: Load Doctors Frame
SystemHandler -> PatientPage : Show Doctors Frame
SystemHandler-> Administrator:getUnverifiedDoctors()
Administrator -->SystemHandler: Unverified Doctors
SystemHandler -->PatientPage: Show Unverified Doctors
alt Right image pressed
User -> PatientPage : Right image
PatientPage -> SystemHandler: Verify Doctor
SystemHandler -> Administrator: VerifyDoctor()
Administrator -->SystemHandler:Confirm
SystemHandler-->PatientPage: show info message\n "Doctor Verified"
else Wrong image pressed
User -> PatientPage : Wrong image
PatientPage -> SystemHandler: Ban Doctor
SystemHandler -> Administrator: BanDoctor()
Administrator -->SystemHandler:Confirm
SystemHandler-->PatientPage: show info message\n "Doctor Suspended"
end
PatientPage->PatientPage: Update Doctors Frame

@enduml
